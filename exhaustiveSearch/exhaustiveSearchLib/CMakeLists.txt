cmake_minimum_required(VERSION 3.1)
add_library(exhaustiveSearchLib STATIC exhaustiveSearchLib.cpp exhaustiveSearchLib.h)
target_link_libraries(exhaustiveSearchLib PUBLIC residualConnectivityCommon)
if(NOT WIN32)
	target_link_libraries(exhaustiveSearchLib PRIVATE ${OpenMP_CXX_FLAGS})
endif()

target_compile_options(exhaustiveSearchLib PRIVATE ${OpenMP_CXX_FLAGS})
target_include_directories(exhaustiveSearchLib PUBLIC ${CMAKE_CURRENT_LIST_DIR})

target_compile_definitions(exhaustiveSearchLib PUBLIC BOOST_ALL_NO_LIB BOOST_ALL_DYN_LINK)
target_compile_definitions(exhaustiveSearchLib PRIVATE USE_OPENMP)
if(WIN32)
	set_target_properties(exhaustiveSearchLib PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR} PDB_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
endif()
