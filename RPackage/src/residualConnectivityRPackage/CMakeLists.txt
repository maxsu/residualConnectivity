project(residualConnectivityRPackage)
cmake_minimum_required(VERSION 3.1)
add_library(residualConnectivityRPackage SHARED main.cpp)
target_include_directories(residualConnectivityRPackage PUBLIC "${residualConnectivity_INCLUDE_DIRS}" "${Boost_INCLUDE_DIRS}")
target_link_libraries(residualConnectivityRPackage PUBLIC mpfr)
target_compile_definitions(residualConnectivityRPackage PUBLIC BOOST_ALL_NO_LIB BOOST_ALL_DYN_LINK)
if(WIN32)
	target_link_libraries(residualConnectivityRPackage PUBLIC "${residualConnectivity_LIBRARY_DIR}/${CMAKE_BUILD_TYPE}/residualConnectivityCommon.lib" mpir)
else()
	target_link_libraries(residualConnectivityRPackage PUBLIC "${residualConnectivity_LIBRARIES}")
	target_compile_features(residualConnectivityRPackage PUBLIC cxx_rvalue_references)
endif()
target_link_libraries(residualConnectivityRPackage PUBLIC ${Boost_LIBRARIES})

install(TARGETS residualConnectivityRPackage RUNTIME DESTINATION . LIBRARY DESTINATION . COMPONENT Runtime)
